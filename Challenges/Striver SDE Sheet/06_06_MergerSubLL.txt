def flatten(root):
    #Your code here
    
    curr = root
    while curr.next : 
        temp = curr.next
        curr.next = curr.next.next
        
        if temp.data < curr.data : 
            curr.data, temp.data = temp.data, curr.data
        
        prev = None
        p1 = curr
        p2 = temp
        
        while p1 and p2 :
            if p1.data <= p2.data :
                prev = p1
                p1 = p1.bottom
            else : 
                tmp = p2
                p2 = p2.bottom
                prev.bottom = tmp
                prev = prev.bottom
                prev.bottom = p1
        
        if p2 : 
            prev.bottom = p2
    
    curr = root
    while curr : 
        curr.next = curr.bottom
        curr = curr.next
    
    return root
