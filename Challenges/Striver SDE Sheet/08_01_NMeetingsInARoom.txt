class Solution:
    
    #Function to find the maximum number of meetings that can
    #be performed in a meeting room.
    def maximumMeetings(self,n,start,end):
        # code here
        
        if n==0 : 
            return 0
            
        meetings = [(start[i], end[i]) for i in range(n)]
        meetings = sorted(meetings, key = lambda x : x[1])
        
        tasks = 1
        point = meetings[0][1]
        for i in range(1, n) :
            if meetings[i][0] > point : 
                point = meetings[i][1]
                tasks += 1
        
        return tasks
