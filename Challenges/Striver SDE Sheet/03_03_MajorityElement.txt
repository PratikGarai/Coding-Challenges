class Solution:
    def majorityElement(self, nums: List[int]) -> int:
        
        mx = nums[0]
        count = 1
        l = len(nums)
        for i in range(1, l):
            if nums[i]==mx : 
                count += 1
            else : 
                count -=1
            if count==0 :
                mx = nums[i]
                count = 1
        return mx
