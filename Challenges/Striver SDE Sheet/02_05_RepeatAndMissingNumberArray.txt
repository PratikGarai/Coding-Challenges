class Solution:
    # @param A : tuple of integers
    # @return a list of integers
    def repeatedNumber(self, A):
        
        l = len(A)
        s = set({})
        
        rep = -1
        for i in A : 
            if i in s : 
                rep = i
                break
            else : 
                s.add(i)
                
        diff = (l*(l+1)/2) - sum(A)
        return [rep, rep+diff]
