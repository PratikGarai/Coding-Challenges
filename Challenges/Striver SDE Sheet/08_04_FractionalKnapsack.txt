class Solution:    
    #Function to get the maximum total value in the knapsack.
    def fractionalknapsack(self, W,Items,n):
        
        # code here
        output = 0
        Items = sorted(Items, key = lambda x : (x.value/x.weight), reverse = True)
        ind = 0
        while W > 0 and ind < n:
            it = Items[ind]
            if it.weight <= W :
                W -= it.weight
                output += it.value
            else : 
                output += (W/it.weight)*it.value
                W = 0
            ind += 1
            
        return output
