class Solution:
    
	def subsetSums(self, arr, N):
		# code here
		
		ans = []
		
		def recur(ind, sm) :
		    if ind == N :
		        ans.append(sm)
		        return
		    recur(ind+1, sm)
		    recur(ind+1, sm+arr[ind])
		   
		recur(0, 0)
		return ans
