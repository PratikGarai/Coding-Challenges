class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        
        nums = sorted(nums)
        ans = set({})
        l = len(nums)
        def recur(ind, a) :
            if ind==l : 
                ans.add(a)
                return
            recur(ind+1, a)
            recur(ind+1, (*a, nums[ind]))
        
        recur(0, ())
        return list(ans)
