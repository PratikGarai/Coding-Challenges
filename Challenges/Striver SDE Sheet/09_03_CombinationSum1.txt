class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        l = len(candidates)
        ans = set({})
        def recur(ind, sm, arr) :
            if sm > target : 
                return
            if sm == target : 
                ans.add(arr)
                return
            if ind==l :
                return
            recur(ind, sm+candidates[ind], (*arr, candidates[ind]))
            recur(ind+1, sm, arr)
            recur(ind+1, sm+candidates[ind], (*arr, candidates[ind]))
        
        recur(0, 0, ())
        return ans
