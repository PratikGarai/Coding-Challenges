class Solution:
    def getIntersectionNode(self, headA: ListNode, headB: ListNode) -> Optional[ListNode]:
        
        l1 = 0
        curr = headA
        while curr : 
            l1 += 1
            curr = curr.next
        
        l2 = 0
        curr = headB
        while curr :
            l2 += 1
            curr = curr.next
        
        if l1>l2 : 
            for i in range(l1-l2) :
                headA = headA.next
        
        if l2>l1 : 
            for i in range(l2-l1) :
                headB = headB.next
        
        while headA and headB : 
            if headA == headB : 
                return headA
            headA = headA.next
            headB = headB.next
        
        return None
