class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:

        l = len(nums)
        nums = sorted([(nums[i], i) for i in range(l)])
        f = 0
        r = l-1
        
        while f < r : 
            if (nums[f][0] + nums[r][0]) == target : 
                return [nums[f][1], nums[r][1]]
            if (nums[f][0] + nums[r][0]) < target : 
                f += 1
            else :
                r -= 1
