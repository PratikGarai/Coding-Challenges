class Solution:
    def maxLen(self, n, arr):
        #Code here
        
        sm = 0
        mx = 0
        
        d = {}
        
        for i in range(n) :
            sm += arr[i]
            
            if sm == 0 : 
              mx = max(mx, i+1)
             
            else :
                if sm not in d : 
                    d[sm] = i
                mx = max(mx, i-d[sm])
        
        return mx
