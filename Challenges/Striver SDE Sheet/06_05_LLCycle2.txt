class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        if not head : 
            return None 
        
        slow = head
        fast = head
        
        while True : 
            slow = slow.next
            fast = fast.next
            
            if not fast : 
                return None  
            
            fast = fast.next
            if not fast : 
                return None
            
            if slow==fast : 
                break
        
        slow = head
        while slow != fast : 
            slow = slow.next
            fast = fast.next
        return slow
