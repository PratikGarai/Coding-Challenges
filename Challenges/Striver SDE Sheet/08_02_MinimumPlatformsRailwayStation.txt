class Solution:    
    #Function to find the minimum number of platforms required at the
    #railway station such that no train waits.
    def minimumPlatform(self,n,arr,dep):
        # code here
        
        if n==0 :
            return 0
            
        temp1 = [0 for i in range(2400)]
        temp2 = [0 for i in range(2400)]
        for i in range(n) : 
            temp1[arr[i]] += 1
            temp2[dep[i]] += 1
        
        platforms = 0
        trains = 0
        for i in range(2400) :
            trains += temp1[i]
            platforms = max(platforms, trains)
            trains -= temp2[i]
        return platforms
