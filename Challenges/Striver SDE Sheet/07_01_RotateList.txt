class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        
        l = 0
        curr = head
        while curr : 
            curr = curr.next
            l += 1
        
        if l <= 1 :
            return head
        
        k = k%l
        if k==0 :
            return head
        
        curr = head
        for i in range(l-k-1) :
            curr = curr.next
        
        ans = curr.next
        tail = ans     
        curr.next = None
        while tail.next : 
            tail = tail.next
        tail.next = head
        return ans
