class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        
        last = -1
        ans = 0
        c = 0
        for i in nums :
            if last != 1 and i == 1 : 
                c = 1
                last = 1
            elif i == 1 :
                c += 1
            else : 
                c = 0
            ans = max(ans, c)
        
        return ans
