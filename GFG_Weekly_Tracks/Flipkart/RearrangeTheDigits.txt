class Solution:
    def smallestnum (self,N):
        # code here 
        if len(N)==1:
            return N
        s = []
        c = 0
        for i in N :
            if i=='0':
                c+=1
            else :
                s.append(i)
        s = sorted(s)
        
        return s[0]+"0"*c+''.join(s[1:])
