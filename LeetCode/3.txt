class Solution:
    def lengthOfLongestSubstring(self, s: str) -> int:
        
        last_repeated_char = -1
        chars = [-1]*256
        l = len(s)
        max_len = 0
        
        for j in range(l):    
            last_repeated_char = max(last_repeated_char, chars[ord(s[j])]+1)
            max_len = max(max_len, j-last_repeated_char+1)
            chars[ord(s[j])] = j
        
        return max_len
