class Solution:
    def countPrimes(self, n: int) -> int:
        
        if n<=2:
            return 0
        
        l = [True for i in range(n)]
        count = 0
        l[2] = True
        
        for i in range(2,n):
            if l[i]:
                count += 1
                
            mul = 2
            while(i*mul<n):
                l[i*mul] = False
                mul += 1
        
        return count
